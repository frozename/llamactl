# UI E2E + smoke test suite — implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a three-tier UI test suite for the Beacon Electron app — a fast PR-blocking Tier A (registry-driven module smoke + bespoke shell smokes), a nightly Tier B (5 surviving flow scripts), and a nightly Tier C (existing cross-repo schema gate, retriggered).

**Architecture:** Tier A is one harness (`tests/ui-flows/tier-a-modules.ts`) that loops `APP_MODULES` and exercises a registry-declared `smokeAffordance` per module, plus 6 hand-written shell smokes under `tests/ui-flows/shell/`. Tier B keeps 5 existing flows (delete 8, repair 5). Tier C is one workflow trigger change. All UI-driving runs on macOS via `electron-mcp` + the existing `pilot-driver.ts` JSON-RPC primitive.

**Tech Stack:** TypeScript, Bun (test runner + script runtime), Electron 41, electron-mcp-server (frozename/electron-mcp, pinned per ui-audit.yml), GitHub Actions on `macos-latest`, `LLAMACTL_TEST_PROFILE=1` for hermetic state.

**Spec:** `docs/superpowers/specs/2026-04-25-ui-e2e-smoke-design.md`

```ts
#!/usr/bin/env -S bun
/**
 * Tier-A module smoke. Navigates to every entry in APP_MODULES via
 * the command palette and asserts the module's smokeAffordance is
 * visible. Halt-on-first-failure. See
 * docs/superpowers/specs/2026-04-25-ui-e2e-smoke-design.md.
 */
import { McpClient, launchElectron, screenshotOnFailure } from './pilot-driver';
import { APP_MODULES } from '../../packages/app/src/modules/registry';

interface ModuleResult {
  id: string;
  affordance: string;
  ok: boolean;
  error?: string;
}

async function main(): Promise<void> {
  const mcp = await launchElectron();
  const results: ModuleResult[] = [];
  try {
    for (const m of APP_MODULES) {
      const r = await runOne(mcp, m.id, m.labelKey, m.smokeAffordance);
      results.push(r);
      if (!r.ok) {
        await screenshotOnFailure(mcp, `tier-a-${m.id}-fail.png`);
        break; // halt-on-first-failure
      }
    }
  } finally {
    await mcp.close();
  }
  const failed = results.find((r) => !r.ok);
  if (failed) {
    console.error(`Tier A FAILED at module ${failed.id}: ${failed.error}`);
    process.exit(1);
  }
  console.log(`Tier A passed: ${results.length}/${APP_MODULES.length} modules`);
}

async function runOne(
  mcp: McpClient,
  id: string,
  labelKey: string,
  affordance: string,
): Promise<ModuleResult> {
  try {
    // Open command palette
    await mcp.send('input.keyChord', { keys: ['Meta', 'Shift', 'KeyP'] });
    await mcp.send('input.type', { text: labelKey });
    await mcp.send('input.key', { key: 'Enter' });

    // Wait for the affordance
    await mcp.send('dom.waitFor', {
      selector: `[data-testid="${affordance}"]`,
      timeoutMs: 5000,
    });

    // Sanity: no error boundary
    const errorBoundaryPresent = await mcp.send('dom.exists', {
      selector: '[data-testid="beacon-error-boundary"]',
    });
    if (errorBoundaryPresent.value) {
      return { id, affordance, ok: false, error: 'react-error-boundary rendered' };
    }

    // Console-error sweep since the start of this iteration
    const consoleErrors = await mcp.send('console.errorsSinceMark', {});
    if (Array.isArray(consoleErrors.value) && consoleErrors.value.length > 0) {
      return { id, affordance, ok: false, error: `console errors: ${JSON.stringify(consoleErrors.value)}` };
    }

    // Reset for next iteration
    await mcp.send('app.evaluate', { script: 'window.useTabStore?.getState().closeAll(false)' });
    return { id, affordance, ok: true };
  } catch (err) {
    return { id, affordance, ok: false, error: String(err) };
  }
}

await main();
```

Note: this assumes electron-mcp's tools surface `input.keyChord`, `input.type`, `input.key`, `dom.waitFor`, `dom.exists`, `console.errorsSinceMark`, `app.evaluate`. If any name doesn't match the actual tool list, adjust to the closest equivalent and document the rename in the commit message — `pilot-driver.ts`'s existing flow scripts use these names today.

#!/usr/bin/env -S bun
/**
 * Shell smoke: cycle through all four themes via ThemeOrbs; assert
 * the [data-theme] attr on <html> flips and persists across reload.
 */
import { McpClient, launchElectron } from '../pilot-driver';

const THEMES = ['sirius', 'ember', 'clinical', 'scrubs'] as const;

async function main(): Promise<void> {
  const mcp = await launchElectron();
  try {
    for (const t of THEMES) {
      await mcp.send('dom.click', { selector: `[data-testid="theme-orb-${t}"]` });
      const attr = await mcp.send('dom.getAttribute', {
        selector: 'html',
        name: 'data-theme',
      });
      if (attr.value !== t) throw new Error(`Theme switch failed: expected ${t}, got ${attr.value}`);
    }
    // Persist last switch (`scrubs`) across reload.
    await mcp.send('app.reload', {});
    const after = await mcp.send('dom.getAttribute', { selector: 'html', name: 'data-theme' });
    if (after.value !== 'scrubs') throw new Error(`Theme did not persist: got ${after.value}`);
    // beacon-theme is the localStorage key (zustand persist name).
    const ls = await mcp.send('app.evaluate', { script: `localStorage.getItem('beacon-theme')` });
    if (!String(ls.value).includes('"version":2')) throw new Error('Theme store version not v2');
    console.log('theme-switch: ok');
  } finally {
    await mcp.close();
  }
}

await main();
```

#!/usr/bin/env -S bun
/**
 * Shell smoke: open three modules, reorder, close middle, assert
 * activeKey fallback to right-then-left, pin one, close-others,
 * pinned survives.
 */
import { launchElectron } from '../pilot-driver';

async function main(): Promise<void> {
  const mcp = await launchElectron();
  try {
    // Open 3 modules via command palette
    for (const id of ['dashboard', 'chat', 'logs']) {
      await mcp.send('app.evaluate', {
        script: `window.useTabStore.getState().open({ tabKey: 'module:${id}', title: '${id}', kind: 'module', openedAt: Date.now() })`,
      });
    }

    // Move middle (chat) to programmatically index 0 — drag is flake-prone in CI;
    // we cover the move() reducer here, drag in a tier-B flow if needed.
    await mcp.send('app.evaluate', { script: `window.useTabStore.getState().move('module:chat', 0)` });

    // Close middle (now logs is in slot 2)
    await mcp.send('app.evaluate', { script: `window.useTabStore.getState().close('module:dashboard')` });
    const active1 = await mcp.send('app.evaluate', { script: 'window.useTabStore.getState().activeKey' });
    // After closing dashboard at index 1, right-neighbour is logs.
    if (active1.value !== 'module:logs') throw new Error(`expected module:logs active after close, got ${active1.value}`);

    // Pin chat, close-others, chat survives
    await mcp.send('app.evaluate', { script: `window.useTabStore.getState().pin('module:chat')` });
    await mcp.send('app.evaluate', { script: `window.useTabStore.getState().closeOthers('module:logs')` });
    const tabs = await mcp.send('app.evaluate', { script: 'window.useTabStore.getState().tabs.map(t => t.tabKey)' });
    const ids = tabs.value as string[];
    if (!ids.includes('module:chat')) throw new Error(`pinned chat tab did not survive: ${JSON.stringify(ids)}`);

    console.log('tab-bar: ok');
  } finally {
    await mcp.close();
  }
}

await main();
```

#!/usr/bin/env -S bun
/**
 * Shell smoke: open every module and rail view, then sweep the DOM
 * for any [data-testid="beacon-error-boundary"]. Belt-and-suspenders
 * — the per-module assertion in tier-a-modules.ts catches the
 * intra-module case; this catches a stale boundary lingering after
 * navigation transitions.
 */
import { launchElectron } from '../pilot-driver';
import { APP_MODULES } from '../../packages/app/src/modules/registry';

async function main(): Promise<void> {
  const mcp = await launchElectron();
  try {
    for (const m of APP_MODULES) {
      await mcp.send('app.evaluate', {
        script: `window.useTabStore.getState().open({ tabKey: 'module:${m.id}', title: '${m.labelKey}', kind: 'module', openedAt: Date.now() })`,
      });
    }
    const present = await mcp.send('dom.exists', { selector: '[data-testid="beacon-error-boundary"]' });
    if (present.value) throw new Error('beacon-error-boundary present after sweep');
    console.log('error-boundary: ok');
  } finally {
    await mcp.close();
  }
}

await main();
